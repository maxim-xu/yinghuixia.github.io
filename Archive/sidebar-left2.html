			// let's make a list to store {42, -3, 17, 99}
			ListNode* node = new ListNode();
			node->data = 42;
			node->next = new ListNode();
			node->next->data = -3;
			node->next->next = new ListNode();
			node->next->next->data = 17;
			node->next->next->next = new ListNode();
			node->next->next->next->data = 99;
			node->next->next->next->next = NULL;
	
			// now let's change from:
			//   node [ 42 ] -> [ -3 ] -> [ 17 ] -> [ 99 ] /
			// to:
			//   node [ 51 ] -> [ 42 ] -> [ -3 ] -> [ 17 ] -> [ 99 ] /
	
			ListNode* newStuff = new ListNode();
			newStuff->data = 51;
			newStuff->next = node;
			node = newStuff;
	
			//             [ 42 ] -> [ -3 ] -> [ 17 ] -> [ 99 ] /
			//             ^
			//     node    |        
			// newStuff [ 51 ] 

			return 0;
		}
		
		
		


		/*
		 * CS 106B, Marty Stepp
		 * This file declares a "ListNode" structure to represent a single element
		 * of data in a list and a pointer to another node.
		 */

		#ifndef _listnode_h
		#define _listnode_h

		/*
		 * The internal structure representing a single node of data.
		 */
		struct ListNode {
			int data;         // element stored in each node
			ListNode* next;   // pointer to the next node (NULL if none)
	
			ListNode(int d = 0, ListNode* n = NULL) {
				data = d;
				next = n;
			}
		};

		#endif
